-- ============================================
-- CORE TABLES
-- ============================================

-- Students table
CREATE TABLE students (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  phone TEXT,
  name TEXT NOT NULL,
  year INTEGER CHECK (year BETWEEN 1 AND 5),
  semester INTEGER CHECK (semester BETWEEN 1 AND 10),
  department TEXT NOT NULL,
  degree TEXT NOT NULL,
  roll_number TEXT,
  batch TEXT,
  target_cgpa DECIMAL(3,2),
  profile_picture TEXT,
  bio TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Subjects/Courses table
CREATE TABLE subjects (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  code TEXT,
  credits INTEGER DEFAULT 3 CHECK (credits BETWEEN 1 AND 10),
  professor TEXT,
  professor_email TEXT,
  room TEXT,
  schedule JSONB, -- {mon: "10:00-11:00", wed: "10:00-11:00"}
  total_classes INTEGER DEFAULT 0,
  color TEXT, -- For UI categorization
  icon TEXT,
  semester INTEGER,
  is_archived BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Attendance records
CREATE TABLE attendance (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE CASCADE,
  date DATE NOT NULL DEFAULT CURRENT_DATE,
  status TEXT NOT NULL CHECK (status IN ('present', 'absent', 'holiday', 'medical', 'cancelled')),
  location POINT, -- GPS coordinates
  notes TEXT,
  marked_at TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(subject_id, date)
);

-- Grades table
CREATE TABLE grades (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE CASCADE,
  exam_type TEXT CHECK (exam_type IN ('mid', 'end', 'assignment', 'quiz', 'lab', 'project')),
  marks_obtained DECIMAL(5,2) NOT NULL,
  total_marks DECIMAL(5,2) NOT NULL,
  weightage DECIMAL(3,2), -- 0.3 for mid-sem, 0.7 for end-sem
  grade_letter TEXT, -- A+, A, B+, etc.
  semester INTEGER,
  exam_date DATE,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Assignments table
CREATE TABLE assignments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE SET NULL,
  title TEXT NOT NULL,
  description TEXT,
  type TEXT CHECK (type IN ('assignment', 'lab', 'project', 'test', 'misc')),
  due_date TIMESTAMPTZ NOT NULL,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'in_progress', 'completed', 'submitted', 'overdue')),
  priority TEXT DEFAULT 'medium' CHECK (priority IN ('low', 'medium', 'high', 'urgent')),
  submission_url TEXT,
  attachments JSONB, -- Array of file URLs
  reminder_sent BOOLEAN DEFAULT FALSE,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Exams table
CREATE TABLE exams (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE CASCADE,
  exam_name TEXT NOT NULL,
  exam_type TEXT CHECK (exam_type IN ('mid', 'end', 'surprise', 'viva')),
  exam_date DATE NOT NULL,
  exam_time TIME,
  duration INTEGER, -- in minutes
  room TEXT,
  syllabus JSONB, -- Array of topics
  notes TEXT,
  timetable_url TEXT,
  is_completed BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Documents table
CREATE TABLE documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  description TEXT,
  category TEXT CHECK (category IN ('id', 'certificate', 'receipt', 'medical', 'form', 'misc')),
  file_url TEXT NOT NULL,
  file_type TEXT, -- application/pdf, image/jpeg
  file_size INTEGER,
  thumbnail_url TEXT,
  metadata JSONB, -- Extracted text, dates, etc.
  tags TEXT[],
  expiry_date DATE,
  is_favorite BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- ============================================
-- SOCIAL & COLLABORATIVE TABLES
-- ============================================

-- Study groups
CREATE TABLE study_groups (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  description TEXT,
  creator_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE SET NULL,
  max_members INTEGER,
  is_public BOOLEAN DEFAULT TRUE,
  meeting_link TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Group members
CREATE TABLE group_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  group_id UUID REFERENCES study_groups(id) ON DELETE CASCADE,
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  role TEXT DEFAULT 'member' CHECK (role IN ('admin', 'moderator', 'member')),
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(group_id, student_id)
);

-- Shared notes
CREATE TABLE shared_notes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id UUID REFERENCES students(id) ON DELETE CASCADE,
  subject_id UUID REFERENCES subjects(id) ON DELETE SET NULL,
  title TEXT NOT NULL,
  content TEXT,
  file_url TEXT,
  is_public BOOLEAN DEFAULT TRUE,
  downloads INTEGER DEFAULT 0,
  likes INTEGER DEFAULT 0,
  tags TEXT[],
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Forum threads
CREATE TABLE forum_threads (

-- Function to calculate attendance percentage
CREATE OR REPLACE FUNCTION calculate_attendance_percentage(
  p_student_id UUID,
  p_subject_id UUID
) RETURNS DECIMAL AS $$
DECLARE
  total_classes INTEGER;
  present_classes INTEGER;
BEGIN
  SELECT COUNT(*) INTO total_classes
  FROM attendance
  WHERE student_id = p_student_id 
    AND subject_id = p_subject_id
    AND status IN ('present', 'absent');
  
  SELECT COUNT(*) INTO present_classes
  FROM attendance
  WHERE student_id = p_student_id 
    AND subject_id = p_subject_id
    AND status = 'present';
  
  IF total_classes = 0 THEN
    RETURN 0;
  ELSE
    RETURN (present_classes::DECIMAL / total_classes * 100);
  END IF;
END;
$$ LANGUAGE plpgsql;

-- Function to calculate GPA
CREATE OR REPLACE FUNCTION calculate_gpa(
  p_student_id UUID,
  p_semester INTEGER DEFAULT NULL
) RETURNS DECIMAL AS $$
DECLARE
  total_credits INTEGER;
  quality_points DECIMAL;
BEGIN
  SELECT 
    SUM(s.credits),
    SUM((g.marks_obtained / g.total_marks * 10) * s.credits)
  INTO total_credits, quality_points
  FROM grades g
  JOIN subjects s ON s.id = g.subject_id
  WHERE g.student_id = p_student_id
    AND (p_semester IS NULL OR g.semester = p_semester);
  
  IF total_credits = 0 THEN
    RETURN 0;
  ELSE
    RETURN (quality_points / total_credits);
  END IF;
END;
$$ LANGUAGE plpgsql;

## ðŸš¨ FAILURE MODES & PREVENTION

### FAILURE MODE 1: "Feature Creep Death Spiral"

**Symptoms:**
- Adding 5+ features per sprint
- Nobody uses 80% of features
- App becomes slow and confusing
- Users say "too complicated"

**Prevention:**
- **Strict 8-feature limit** for Phase 1
- Kill any feature with <20% adoption after 1 month
- User testing every 2 weeks: "Can you mark attendance in 10 seconds?"

**Recovery:**
- Emergency feature purge
- Revert to Phase 1 simplicity
- Rebuild trust with "back to basics" messaging

***

### FAILURE MODE 2: "Premature Scaling"

**Symptoms:**
- Optimizing for 100K users when you have 100
- Building infrastructure before users
- Spending on ads before product-market fit
- Hiring team before revenue

**Prevention:**
- **No marketing spend** until 50%+ Week 2 retention
- Manual onboarding for first 100 users
- Use free tiers (Supabase, Vercel) as long as possible
- Solo founder until â‚¹5L MRR

**Recovery:**
- Cut all paid acquisition immediately
- Focus 100% on retention of existing users
- Build features existing users **beg for**

***

### FAILURE MODE 3: "Wrong Users"

**Symptoms:**
- Users sign up but never return
- Low engagement despite high sign-ups
- Users ask for features outside core use case
- Churn after 1 week

**Prevention:**
- **Hyper-specific targeting**: 3rd year engineering students in India
- Recruit beta users from **one college** first
- Onboarding asks: "Are you a current college student?"
- Reject users outside target demo (politely)

**Recovery:**
- Survey churned users: "Why did you stop?"
- Rebuild onboarding for **right persona**
- Refund wrong-fit customers, learn why they churned

***

### FAILURE MODE 4: "Competition Paralysis"

**Symptoms:**
- Copying features from competitors
- Losing unique positioning
- Feature parity obsession
- "They launched X, we need X too!"

**Prevention:**
- **Ignore competitors** during Phase 1-2
- Focus on **your users' problems**, not competitor features
- Unique value prop: Student-owned data, not institution-controlled

**Recovery:**
- Delete copied features
- Talk to 20 users: "What would you miss if this app disappeared?"
- Double down on **that one thing**

***

### FAILURE MODE 5: "Monetization Too Early"

**Symptoms:**
- Introducing paid tiers before proving value
- Users angry about paywalls
- Low conversion (<1%)
- Negative reviews mentioning pricing

**Prevention:**
- **Zero monetization** until 1,000 active users
- Free tier must solve 80% of use cases
- Premium must be **10x better**, not basic features locked
- 14-day free trial, no credit card

**Recovery:**
- Make controversial features free again
- Apologize publicly
- Rebuild pricing based on **willingness to pay** surveys

***

### FAILURE MODE 6: "Technical Debt Bankruptcy"

**Symptoms:**
- New features take 2x longer to build
- Bugs increase exponentially
- Crashes on production
- Developers say "need to rewrite"

**Prevention:**
- **20% time** for refactoring every sprint
- Test coverage >80%
- Code reviews mandatory
- No "quick hacks" in production

**Recovery:**
- Feature freeze for 1 month
- Pay down technical debt systematically
- Rebuild on solid foundation
